const Web3 = require("web3");

const provider = '';//یک نود رو شبکه اتریم به رایگان میتونید توی quicknode بسازید 
const Web3Client = new Web3(new Web3.providers.HttpProvider(provider));

const minABI = [
    {
        constant: true,
        inputs: [{name: "_owner", type: "address"}],
        name: "balanceOf",
        outputs: [{name: "balance", type: "uint256"}],
        type: "function",
    },
];
const tokenAddress = "";//آدرس قرارداد هوشمند توکن یا NFT
const walletAddress = "";// آدرس کیف پولی که میخواید چک بشه 

const contract = new Web3Client.eth.Contract(minABI, tokenAddress);

async function getBalance() {
    const result = await contract.methods.balanceOf(walletAddress).call();
    const format = Web3Client.utils.fromWei(result);

    console.log(format);
    if (format > 0) {
        console.info('NFT exist');
    } else {
        console.warn('NFT not exist');
    }
}

getBalance();
